import { getSession } from "./utils/session.js";
app.get("/me", (req, res) => {
  const sessionId = req.cookies.sessionId;
  const session = getSession(sessionId);
  if (!session) {
    return res.status(401).json({ message: "Not authenticated" });
  }
  res.json({
    userId: session.userId
  });
});

//HELMET

Som jeg forstår det, så beskytter helmet dine headers, som besværliggør angreb. Man kan med en konfiguration gøre det totalt sikkert ift hvad din side egentlig må loade.
Gemini: Helmet er en middleware til Express (og andre Node.js-frameworks), der automatisk opsætter forskellige HTTP-sikkerhedshoveder (HTTP security headers).
        I stedet for at du manuelt skal konfigurere hvert eneste sikkerhedshoved i din app.js eller server.js, gør Helmet det for dig med én simpel kommando.
        Hvad er CSP (Content Security Policy)? - CSP er et HTTP-hoved, der giver dig mulighed for at kontrollere, hvilke ressourcer (scripts, styles, billeder, osv.) din browser har tilladelse til at loade på din webside. Det er et af de stærkeste forsvar mod Cross-Site Scripting (XSS) og data-injiceringsangreb.
        Hvordan det virker: Du sender en CSP-politik fra din server til browseren. Hvis siden forsøger at loade et script fra en domæne, som ikke er på din "white-list" (tilladelsesliste), vil browseren blokere det.
        Eksempel: En politik kunne sige: "Tillad kun scripts fra mit eget domæne og fra trusted-scripts.com". Alt andet bliver afvist.

// Vi bruger en konkret CSP-konfiguration i stedet for kun standarden.
app.use(
  helmet({
    contentSecurityPolicy: {
      directives: {
        // Tillad kun ressourcer fra samme origin som udgangspunkt.
        defaultSrc: ["'self'"],

        // Scripts må kun komme fra samme origin.
        // Vi tillader ikke inline scripts her.
        scriptSrc: ["'self'"],

        // Styles må komme fra samme origin.
        // Hvis frontend senere bruger inline styles eller style attributes,
        // kan vi blive nødt til at justere strategien dér.
        styleSrc: ["'self'"],

        // Billeder må komme fra samme origin og data: URL'er.
        // data: er ofte praktisk til små inline assets.
        imgSrc: ["'self'", "data:"],

        // API-kald / fetch / XHR må kun gå til samme origin.
        // Hvis frontend ligger et andet sted i dev, kan vi justere senere.
        connectSrc: ["'self'"],

        // Blokér object/embed som defense-in-depth.
        objectSrc: ["'none'"],

        // Begræns base-tag manipulation.
        baseUri: ["'self'"],

        // Beskyt mod clickjacking via frame ancestors.
        frameAncestors: ["'none'"],

        // Formularer må kun submitte til samme origin.
        formAction: ["'self'"]
      }
    },

    // Slå ikke CSP fra.
    // Vi vil faktisk have den aktiv.
    crossOriginEmbedderPolicy: false
  })
);

// Seperation of concerns

Serveren er bygget modulært, så anvarsområder opdeles og appen kan testes og vedligeholdes nemmere.
"controllers" er processer der på en måde tager brugeren/klienten/etEllerAndet fra en status til en anden. De "gør" egentlig ikke noget. Derfor er register, login, logout, forgot password, reset password, info om mig /me, - i auth.controllers.
"auth" står for authentication - altså "hvem er du?"
"middleware" er steps der ligger mellem en url request fra klienten til at en funktion udføres. En dørmand til en graduation, der tjekker om du nu også er den, du siger du er - agtigt.
"validate" returnerer bare om et schema er truthy eller falsy, basically. Så hvis noget ikke passer i format, så siger den nix.
"routes" giver sig selv. auth.routes indikerer at de her ruter handler om athentication.
"utils" er utilities, og antyder "værktøjer", altså funktionaliteter, som resten af projektet kan bruge som værktøjer i sin opbygning.
